    def show_list_for(record, show_column_set)
      show_column_set.map do |column|
        content_tag(:p, :class => 'ponteggio-show') do
          content_tag(:span, column[:name] + ': ', :class => 'label') +
          ((column[:list] == :long) ? '<br />' : '') + 
          value_for(record, column)
        end
      end * "\n"
    end
